namespace = SLMod_events_crowded_navy_namespace

# Sailors usage 
country_event = {
	id = SLMod_events_crowded_navy_namespace.1
	title = SLMod_events_crowded_navy_namespace.1_title
	desc = SLMod_events_crowded_navy_namespace.1_desc
	picture = CORRUPTION_eventPicture
	hidden = no

	trigger {
		sailors_percentage = 0.90
	}
	
	mean_time_to_happen = {
		days = 120
	}
	
	# Boost privateer_efficiency
	option {
		name = SLMod_events_crowded_navy_namespace.1_option_1
		highlight = yes
		trigger {
			has_privateers = yes
		}
		add_country_modifier {
			name = efficient_privateers_country_modifier
			duration = 365
			hidden = no
		}
		# Trade 15 months of sailors for 12 months of bonus
		add_yearly_sailors = -1.25
	}
	
	# Boost navy tradition
	option {
		name = SLMod_events_crowded_navy_namespace.1_option_2
		highlight = yes
		trigger {
			NOT {
				navy_tradition = 90
			}
		}
		add_country_modifier {
			name = slow_navy_tradition_decay_country_modifier
			duration = 365
			hidden = no
		}
		# Trade 15 months of sailors for 12 months of bonus
		add_yearly_sailors = -1.25
	}

	# Cycle them for manpower
	option {
		name = SLMod_events_crowded_navy_namespace.1_option_3
		highlight = yes
		trigger {
			NOT {
				manpower_percentage = 50
			}
		}
		# Trade 12 months of sailors for 5 months of manpower
		add_yearly_sailors = -1
		add_yearly_manpower = 0.42
	}
	
	# Boost trade
	option {
		name = SLMod_events_crowded_navy_namespace.1_option_4
		highlight = yes
		trigger {
			ruler_has_personality = entrepreneur_personality
		}
		add_country_modifier {
			name = trade_steering_country_modifier
			duration = 365
			hidden = no
		}
		# Trade 15 months of sailors for 12 months of bonus
		add_yearly_sailors = -1.25
	}
	
	# Default
	option {
		name = SLMod_events_crowded_navy_namespace.1_option_99
		add_country_modifier {
			name = crowded_navy_country_modifier
			duration = 365
			hidden = no
		}
	}
}
